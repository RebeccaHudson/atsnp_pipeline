#The commands in here will  create an index for the gencode gene names; 
#then create the mapping for that index.

curl -XPUT 'atsnp-db2:9200/gencode_genes'  #creates this index. 
                                           #why not let gencode genes be their own index?

#this is in proper form, and can be pasted directly into the terminal.
curl -XPUT 'atsnp-db2:9200/gencode_genes/_mapping/gencode_gene_symbols?pretty' -H 'Content-Type: application/json'  -d'
  {"properties" : {
    "chr" : {
      "type" : "string"
    },
    "end_pos" : {
      "type" : "integer"
    },
    "gene_symbol" : {
      "type" : "keyword"
    },
    "start_pos" : {
      "type" : "integer"
    }
  }}'
